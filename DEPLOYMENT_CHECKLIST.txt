【部署步骤】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Linux:
  1. scp -r vita/ user@server:/opt/vita/
  2. ssh user@server
  3. cd /opt/vita
  4. ./deploy_server.sh
  5. python3 -m streamlit run vita.py --server.port=8501

Windows:
  1. 复制 vita 文件夹到服务器
  2. cd C:\vita
  3. deploy_server.bat
  4. python -m streamlit run vita.py --server.port=8501
